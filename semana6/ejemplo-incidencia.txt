Ejemplo de incidencia comunitaria

{
  "tipo": "Iluminacion publica",
  "descripcion": "Una lampara del parque no funciona desde hace varios dias, generando inseguridad en el sector durante las noches."
}

{
  "mensaje": "Incidencia registrada",
  "tipo": "Iluminacion publica",
  "descripcion": "Una lampara del parque no funciona desde hace varios dias, generando inseguridad en el sector durante las noches."
}
